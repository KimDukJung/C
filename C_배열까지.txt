int fact(int n){
    if (n == 0) return 1;
    return n * fact(n-1);
}

int main()
{
    int a = 3;
    int result = fact(a);
    printf("%d! = %d\n", a, result);
    return 0;
}

int main()
{
    int arr[5] = {5, 2, 7, 9, 4};
    int max = arr[0];
    int length = sizeof(arr) / sizeof(arr[0]);
    for(int i = 1; i < length; i++){
        if( max < arr[i])
            max = arr[i];
    }
    printf("%d \n", max);
    printf("%d, %d", sizeof(arr), sizeof(arr[0]));
    return 0;
}

25.
void cng(int a[]){
    int temp = a[0];
    a[0] = a[1];
    a[1] = temp;
}

int main()
{
    int arr[3] = {5, 2, 7};
    printf("변환 전 : %d, %d, %d\n", arr[0], arr[1], arr[2]);
    cng(arr);
    printf("변환 후 : %d, %d, %d\n", arr[0], arr[1], arr[2]);
    return 0;
}
