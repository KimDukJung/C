* 2차원 배열

1.
#include <stdio.h>
int main(){
	int a[2][3] = { {1,2,3},{4,5,6}};
	for(int i =0; i<2; i++){
		for(int j = 0; j<3; j++){
			printf("%d ", a[i][j]);
		}
		printf("\n");
	}
	return 0;
}

#include <stdio.h>
int main(){
	int a[2][3] = { {1, 2 , 3}, {4, 5, 6} };
	printf("%d\n", a == a[0]);
	printf("%d\n", a == &a[0]);
	printf("%d\n", a[1] == *(a+1));
	printf("%d\n", &a[0] == *(a+0));
	return 0;
}
