8. 
#include <stdio.h>
struct Link {
	int v;
	int next;
};
int main(){
	struct Link link[4] = { 10, 1, 30, 2, 50, 3, 60, -1};
	int head = 0;
	int del = 2;
	int prev = 1;
	link[prev].next = link[del].next;
	link[del].next = -1;
	int p = head;
	while( p != -1){
		printf("%d ", link[p].v);
		p = link[p].next;
	}
	return 0;
}

6.
#include <stdio.h>
struct Node {
	int value;
	struct Node * next;
};

int main(){
	struct Node n1 = { 10, NULL};
	struct Node n2 = { 20, NULL};
	struct Node n3 = { 30, NULL};
	
	n1.next = &n2;
	n2.next = &n3;
	struct Node newNode = { 15, NULL };
	newNode.next = n1.next;
	n1.next = &newNode;
	struct Node *p = &n1;
	while( p != NULL){
		printf("%d ", p->value);
		p = p->next;
	}
	return 0;
}

};

1.

#include <stdio.h>
#include <unistd.h>
int main(){
	int a = 0;
	fork();
	a = 1;
	printf("%d\n", a);
	return 0;
}

2.
#include <stdio.h>
#include <unistd.h>
int main(){
	pid_t id;
	int a = 0;
	id = fork();
	if (id > 0){
		a = 1;
		printf("부모 : %d\n", a);
	} else if( id == 0){
		a = 2;
		printf("자식 : %d\n", a);
	} else{
		printf("실패");
	}
	return 0;
}
