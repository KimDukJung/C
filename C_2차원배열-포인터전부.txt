#include <stdio.h>
int main(){
	int a[2][3] = { {1, 2 , 3}, {4, 5, 6} };
	printf("%d, %d\n", a+1, *a+1);
	printf("%d, %d\n", a, *a);
	printf("%d, %d\n", *(a+1), *(*a+1));
	printf("%d, %d, %d\n", a[0], *a[0], a[0]+1);
	printf("%d, %d, %d\n", sizeof(a), sizeof(*a), sizeof(&a));
	printf("%d, %d, %d\n", *a, **a, *(a[0]+1));
	printf("%d, %d, %d\n", *a, *(a+1), *(a+1)+1);
	printf("%d, %d, %d\n", *(a[0]+0), *(*(a+1)+2), *(a+1)+1);
	return 0;
}

1.
#include <stdio.h>
int main(){
	int a=1, b=2, c=3;
	int *pa[3] = {&a, &b, &c};
	printf("%d, %d, %d\n", pa[0], &pa[0][0], *pa);
	printf("%d, %d, %d", *pa[0], pa[0][0], **pa);
	return 0;
}

2. 
#include <stdio.h>
int main(){
	int a=1;
	int *pa = &a;
	int **ppa = &pa;
	printf("%d, %d, %d, %d\n", a, *pa, pa[0], *(pa+0));
	printf("%d, %d, %d\n", &a, pa, &pa[0]);
	printf("%d", **ppa);
	return 0;
}
